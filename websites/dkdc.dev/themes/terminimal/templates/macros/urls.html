{% macro rel_url(path, trailing_slash=false, cachebust=false) %} {%- set url = get_url(path=path, trailing_slash=trailing_slash, cachebust=cachebust) | replace(from=config.base_url, to="") -%} {%- if url is starting_with("//") -%} /{{- url | trim_start_matches(pat="//") -}} {%- else -%} {{- url -}} {%- endif -%} {% endmacro rel_url %}
